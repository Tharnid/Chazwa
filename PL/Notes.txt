Multi Line 
==========

-- single statement does not require a return

foundCustomer = customerList.FirstOrDefault(c =>
{
	Debug.WriteLine(c.LastName);
	return c.CustomerId == customerId; // or whatever
})

Chaining

foundCustomer = customerList.FirstOrDefault(c =>
{
	.Where(c =>
		c.CustomerId == customerId)
	.Skip(1)
	.FirstOrDefault();
})
